006-002_lab_Q Network for Cart Pole___07_1_q_net_cartpole.py.html ''' This code is based on https://github.com/hunkim/DeepRL-Agents ''' import numpy as np import tensorflow as tf from collections import deque import gym # You build cart poloe environment env = gym.make('CartPole-v0') # You define constants defining your neural network learning_rate = 1e-1 # input_size will be 4 (float values) input_size = env.observation_space.shape[0] # output_size will be 2 (left, right) output_size = env.action_space.n X = tf.placeholder(tf.float32, [None, input_size], name="input_x") # weight in first layer # If you use get_variable(), # you can use variable's name, # and xavier_initializer() W1=tf.get_variable( "W1" ,shape=[input_size, output_size] ,initializer=tf.contrib.layers.xavier_initializer()) Qpred = tf.matmul(X, W1) # We need to define the parts of the network needed for learning a policy Y = tf.placeholder(shape=[None, output_size], dtype=tf.float32) # c loss: loss function node loss = tf.reduce_sum(tf.square(Y - Qpred)) # c train: to be trained node train = tf.train.AdamOptimizer(learning_rate=learning_rate).minimize(loss) # Values for q learning max_episodes = 5000 dis = 0.9 step_history = [] # Setting up our environment init = tf.global_variables_initializer() sess = tf.Session() sess.run(init) for episode in range(max_episodes): e = 1. / ((episode / 10) + 1) step_count = 0 state = env.reset() done = False # The Q-Network training while not done: step_count += 1 # state: state you got # You should convert shape of state to make it fitted with your shape of input # then, you'll feed this x x = np.reshape(state, [1, input_size]) # Choose an action by greedily (with e chance of random action) from # the Q-network Q = sess.run(Qpred, feed_dict={X: x}) if np.random.rand(1) < e: action = env.action_space.sample() else: action = np.argmax(Q) # Get new state and reward from environment next_state, reward, done, _ = env.step(action) # Defining target $$$y_{j}$$$ (label y) algorithm: # If done, you use only reward # $$$y_{j}=r_{j}$$$ # If not done, you use reward and next highest Q # $$$y_{j}=r_{j}+\gamma max_{a'} Q(\phi_{j+1},a';\theta)$$$ if done: # If you fall, you hugely mistake, so take -100 reward Q[0, action] = -100 else: # You reshape next_state with [1, input_size] x_next = np.reshape(next_state, [1, input_size]) # Obtain the Q' values by feeding the new state through our network Q_next = sess.run(Qpred, feed_dict={X: x_next}) # What is y label? # $$$y=r+\gamma max Q(s')$$$ # We have 2 actions(left, right) # [[1*2]][[2*1]]=[[1*1]] Q[0, action] = reward + dis * np.max(Q_next) # Train our network using target and predicted Q values on each episode sess.run(train, feed_dict={X: x, Y: Q}) state = next_state step_history.append(step_count) print("Episode: {} steps: {}".format(episode, step_count)) # If last 10's average steps are 500, it's good enough if len(step_history) > 10 and np.mean(step_history[-10:]) > 500: break # You will see if your trained model works well observation = env.reset() reward_sum = 0 while True: env.render() # You create x by reshaping observation (current state) x = np.reshape(observation, [1, input_size]) # With x, you ask how to do to Q Q = sess.run(Qpred, feed_dict={X: x}) action = np.argmax(Q) observation, reward, done, _ = env.step(action) reward_sum += reward if done: print("Total score: {}".format(reward_sum)) break
